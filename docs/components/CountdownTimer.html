      if( isNaN(datetime) )

      }
  
      console.log( datetime + " " + now);
  
      if (datetime < now) {
          var milisec_diff = now - datetime;
      }else{
          var milisec_diff = datetime - now;
      }
      var days = Math.floor(milisec_diff / 1000 / 60 / (60 * 24));
      var date_diff = new Date( milisec_diff );
